pm.test('Status code is 200', function () {
    pm.response.to.have.status(200);
});
pm.test('Response time < 2000ms', function () {
    pm.expect(pm.response.responseTime).to.be.below(2000);
});
pm.test('Address created successfully', function () {
    const jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property('success').that.is.a('boolean');
    pm.expect(jsonData).to.have.property('address_id').that.is.a('string');
    pm.expect(jsonData.success).to.be.true;
});
pm.test('Save address_id', function () {
    const jsonData = pm.response.json();
    pm.collectionVariables.set('address_id', jsonData.address_id);
    pm.expect(jsonData.address_id).to.match(/^addr_\d+$/);
});
pm.test('Content-Type is JSON', function () {
    pm.expect(pm.response.headers.get('Content-Type')).to.include('application/json');
});
