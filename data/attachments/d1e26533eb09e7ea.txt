pm.test('Status code is 200', function () {
    pm.response.to.have.status(200);
});
pm.test('Price calculation has required fields', function () {
    const jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property('total').that.is.a('number');
    pm.expect(jsonData).to.have.property('payable_amount').that.is.a('number');
    pm.expect(jsonData).to.have.property('total_meals').that.is.a('number');
    pm.expect(jsonData).to.have.property('price_per_meal').that.is.a('number');
});
