pm.test('Status code is 200', function () {
    pm.response.to.have.status(200);
});
pm.test('Recharge response has required fields', function () {
    const jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property('success').that.is.a('boolean');
    pm.expect(jsonData.success).to.be.true;
    pm.expect(jsonData).to.have.property('order_id');
    pm.expect(jsonData).to.have.property('amount').that.equals(500);
    pm.collectionVariables.set('order_id', jsonData.order_id);
});
