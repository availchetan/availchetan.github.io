pm.test('Status code is 200', function () {
    pm.response.to.have.status(200);
});
pm.test('Response time < 2000ms', function () {
    pm.expect(pm.response.responseTime).to.be.below(2000);
});
pm.test('Response is an array with at least 1 address', function () {
    const jsonData = pm.response.json();
    pm.expect(jsonData).to.be.an('array');
    pm.expect(jsonData.length).to.be.at.least(1);
});
pm.test('Saved address_id exists in list', function () {
    const jsonData = pm.response.json();
    const savedId = pm.collectionVariables.get('address_id');
    const exists = jsonData.some(addr => addr.id === savedId);
    pm.expect(exists).to.be.true;
});
pm.test('Content-Type is JSON', function () {
    pm.expect(pm.response.headers.get('Content-Type')).to.include('application/json');
});
